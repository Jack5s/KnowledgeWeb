Prevent PostgreSQL Injection

Use parameter to prevent injection

String insertStr = "insert into schemaName.\"tableName\" VALUES (?,?,?,?,?)";
PreparedStatement preparedStatement = ParaSetting.connection.prepareStatement(insertStr);
preparedStatement.setString(1, id);
preparedStatement.setDouble(2, longitude);
preparedStatement.setDouble(3, latitude);
preparedStatement.setString(5, url);
preparedStatement.setTimestamp(6, new Timestamp(calendar.getTimeInMillis()));
preparedStatement.executeUpdate();
preparedStatement.close();
